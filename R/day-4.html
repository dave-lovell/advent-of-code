Task 1: How many points?

scratch <- 
  scratch |> 
  rowwise() |> 
  mutate(wins = have %in% winning) |> 
  group_by(card) |> 
  summarise(winners = sum(wins), .groups = "drop") |> 
  mutate(points = (winners > 0) * 2^(winners - 1))

total_points <- sum(scratch[["points"]])
